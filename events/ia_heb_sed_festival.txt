namespace = ba_egypt_flavor

ba_egypt_flavor.1 = {
	type = country_event
	hidden = yes
	
	immediate = {
		if = {
			limit = {
				religion = kemetic_pantheon
			}
			current_ruler = {
				if = {
					limit = {
						has_variable = years_reigning
					}
					change_variable = {
						name = years_reigning
						add = 1
					}
				}
				else = {
					set_variable = {
						name = years_reigning
						value = 1
					}
				}
			}
		}
	}
}
