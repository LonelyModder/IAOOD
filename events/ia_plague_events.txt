plague_event.2 = { 
	type = country_event
	hidden = yes
	
	immediate = {
		every_province = {
			if = {
				limit = {
					any_neighbor_province = { has_country_modifier = plagued_territory }
				}
				add_province_modifier = {
					name = sick_territory
				}
			}
		}
	}
}
