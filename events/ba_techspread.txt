namespace = ba_techspread

ba_techspread.1 = { #
	type = country_event
	hidden = yes

	trigger = {
		is_tribal = yes
		oratory_tech < 2
	}
	immediate = {
		random_country = {
			limit = {
				government = despotic_monarchy
				OR = {
					exports_to = root
					imports_from = root
				}
			}
			save_scope_as = techtarget
		}
		if = {
			limit = {
				exists = scope:techtarget	
			}
			trigger_event =  {
				id = ba_techspread.2
			}
		}
	}

}
