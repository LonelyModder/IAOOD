	}
	
	lex_gabinia = {
		allow  = {
			num_of_ports >= 1
		}
		modifier = {
			global_export_commerce_modifier = 0.25
		}
	}
	
	lex_caecilia = {
		allow  = {
			num_of_ports >= 1
		}
		modifier = {
			global_import_commerce_modifier = 0.25
		}
		on_enact = {
			if = {
				limit = {
					any_owned_province = {
						NOT = { has_province_modifier = pirate_haven_modifier }
					}
				}
				custom_tooltip = pirate_will_be_placed_tt
				hidden_effect = {
					random_owned_province = {
						limit = {
							is_port = yes
						}
						weight = {
							modifier = {
								factor = 1000
								controller = root
							}
						}
						add_permanent_province_modifier = {
							name = pirate_haven_modifier
							duration = -1
						}
					}
				}
			}
		}
	}
	
	lex_claudia = {
		allow  = {
			num_of_ports >= 1
		}
		modifier = {
			anti_piracy_cb = yes
		}
	}
	
}
