port_assault = {

	enable = yes
	toggle = no
	soundeffect = "event:/SFX/UI/Unit/sfx_ui_unit_port_assualt"


	navy_only = yes


	allow = {
		is_navy = yes
		in_combat = no
		is_moving = no
		is_exiled = no
		custom_tooltip = {
			text = port_assault_explain_tt
			any_sub_unit = {
				sub_unit_type = mega_galley
				subunit_strength_percentage >= 0.95
			}
		}
		trigger_if = {
			limit = {
				exists = commander
			}
			commander = {
				loyalty > 33
			}
		}
		trigger_if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						is_sea = no
						has_owner = yes
						is_port = yes
						fort_level >= 1
						NOT = { controller = root.unit_owner }
						controller = { war_with = root.unit_owner }
						NOT = { has_enemy_army = root.unit_owner }
					}
				}
			}
			unit_location = {
				any_neighbor_province = {
					is_sea = no
					has_owner = yes
					is_port = yes
					fort_level >= 1
					NOT = { controller = root.unit_owner }
					controller = { war_with = root.unit_owner }
					NOT = { has_enemy_army = root.unit_owner }
				}
			}
		}
		trigger_else = {
			custom_tooltip = {
				text = port_assault_no_valid_port
				unit_location = {
					any_neighbor_province = {
						is_sea = no
						has_owner = yes
						is_port = yes
						fort_level >= 1
						NOT = { controller = root.unit_owner }
						controller = { war_with = root.unit_owner }
						NOT = { has_enemy_army = root.unit_owner }
					}
				}
			}
		}
	}
	
	
	
	start_effect = {
		random_sub_unit = {
			limit = {
				sub_unit_type = mega_galley
				subunit_strength_percentage >= 0.95
			}
			add_subunit_strength = -0.70
		}
		if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						is_sea = no
						has_owner = yes
						is_port = yes
						fort_level >= 1
						NOT = { controller = root.unit_owner }
						controller = { war_with = root.unit_owner }
						NOT = { has_enemy_army = root.unit_owner }
					}
				}
			}
			unit_location = {
				random_neighbor_province = {
					limit = {
						is_sea = no
						has_owner = yes
						is_port = yes
						fort_level >= 1
						NOT = { controller = root.unit_owner }
						controller = { war_with = root.unit_owner }
						NOT = { has_enemy_army = root.unit_owner }
					}
					remove_building_level = fortress_building
				}
			}
		}
		else = {
			custom_tooltip = port_assault_explain_effect_tt
		}
	}
	
}
