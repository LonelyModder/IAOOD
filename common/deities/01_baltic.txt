	trigger = {
		polytheistic_check_religion_trigger = yes

			}
			deity:omen_mmedeine = {
				holy_site_deity_check_trigger = yes
			}
		}
	}

	omen = { land_morale_modifier = omen_land_morale_modifier_svalue } #Bonus from Omen

	}
	religion = baltic_pantheon

	}
}
